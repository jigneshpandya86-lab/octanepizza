  <script>
    let globalMenu = [];
    let cart = [];
    let pendingOrder = null;
    let editingOrderId = '';
    let pendingPaymentOrder = null;
    let pendingBillOrderId = null;
    let currentOrders = [];

    // Navigation
    function nav(pageId, btnEl) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
      document.getElementById(pageId).classList.add('active');
      const labels = {
        pos: ['Point of Sale', 'POS'],
        orders: ['Orders'],
        tables: ['Tables'],
        menu: ['Menu'],
        expenses: ['Expenses']
      };
      document.querySelectorAll('.nav-btn').forEach(btn => {
        if ((labels[pageId] || []).includes(btn.textContent.trim())) btn.classList.add('active');
      });
      if (btnEl) btnEl.classList.add('active');
      document.getElementById('topNav').classList.remove('open');
    }

    function toggleMobileMenu() {
      document.getElementById('topNav').classList.toggle('open');
    }

    function showToast(message, type = 'info') {
      const wrap = document.getElementById('toastWrap');
      const toast = document.createElement('div');
      toast.className = `toast ${type}`;
      toast.innerText = message;
      wrap.appendChild(toast);
      setTimeout(() => {
        toast.style.opacity = '0';
        toast.style.transform = 'translateY(8px)';
      }, 2300);
      setTimeout(() => toast.remove(), 2700);
    }

    // Startup
    window.onload = function() {
      loadPosMenu();
      loadTableSetup();
      loadOrders();
    };

    // --- POS FUNCTIONS ---
    function loadPosMenu() {
      google.script.run.withSuccessHandler(function(menu) {
        globalMenu = menu;
        const posDiv = document.getElementById('posMenuList');
        posDiv.innerHTML = '';
        menu.filter(m => m.status === 'Active').forEach(item => {
          posDiv.innerHTML += `<div class="menu-item-btn" onclick="addToCart('${item.id}', '${item.name}', ${item.price})">
            <strong>${item.name}</strong><br>₹${item.price}
          </div>`;
        });
      }).getMenu();
    }

    function addToCart(id, name, price) {
      const existing = cart.find(i => i.id === id);
      if (existing) existing.qty++;
      else cart.push({ id, name, price, qty: 1 });
      renderCart();
    }

    function renderCart() {
      const cartDiv = document.getElementById('cartItems');
      cartDiv.innerHTML = '';
      let total = 0;
      cart.forEach((item, index) => {
        total += item.price * item.qty;
        cartDiv.innerHTML += `<li class="cart-row">
          <span>${item.qty}x ${item.name}</span>
          <span>₹${item.price * item.qty} <button onclick="cart.splice(${index},1); renderCart();" class="cart-remove">✖</button></span>
        </li>`;
      });
      document.getElementById('cartTotal').innerText = total;
    }

    function processOrder() {
      if (cart.length === 0) return showToast('Cart is empty. Add items before booking.', 'info');
      pendingOrder = {
        id: editingOrderId,
        items: cart,
        table: document.getElementById('posTableSelect').value,
        total: document.getElementById('cartTotal').innerText,
        mode: 'Pending',
        customerName: document.getElementById('clientName').value || 'Walk-in',
        customerPhone: document.getElementById('clientPhone').value || '-'
      };
      if (pendingOrder.customerPhone !== '-' && !/^\d{10}$/.test(pendingOrder.customerPhone)) {
        return showToast('Please enter a valid 10-digit mobile number.', 'info');
      }
      document.getElementById('confirmText').innerText = `${editingOrderId ? 'Updating' : 'Saving'} order for Table: ${pendingOrder.table}`;
      document.getElementById('confirmText2').innerText = `${pendingOrder.customerName} (${pendingOrder.customerPhone}) • Total: ₹${pendingOrder.total}`;
      document.getElementById('confirmModal').classList.add('show');
    }

    function closeConfirmModal() {
      pendingOrder = null;
      document.getElementById('confirmModal').classList.remove('show');
    }

    function confirmAndSaveOrder() {
      if (!pendingOrder) return;
      document.getElementById('confirmModal').classList.remove('show');
      const saveBtn = document.getElementById('saveOrderBtn');
      saveBtn.innerText = editingOrderId ? 'Saving Changes...' : 'Saving...';
      saveBtn.disabled = true;
      google.script.run.withSuccessHandler(function(res) {
        cart = [];
        renderCart();
        document.getElementById('clientName').value = '';
        document.getElementById('clientPhone').value = '';
        document.getElementById('saveOrderBtn').innerText = 'Save Order';
        document.getElementById('saveOrderBtn').disabled = false;
        showToast(`Order ${res.id} saved successfully.`, 'success');
        editingOrderId = '';
        pendingOrder = null;
      }).saveOrder(pendingOrder);
    }

    function printReceipt(orderData) {
      const subtotal = Number(orderData.subtotal ?? orderData.total ?? 0);
      const discountPercent = Number(orderData.discountPercent || 0);
      const discountAmount = Number(orderData.discountAmount || 0);
      const finalTotal = Number(orderData.total || 0);
      document.getElementById('printId').innerText = orderData.id;
      document.getElementById('printDate').innerText = orderData.date;
      document.getElementById('printTable').innerText = orderData.table;
      document.getElementById('printSubtotal').innerText = subtotal.toFixed(2);
      document.getElementById('printTotal').innerText = finalTotal.toFixed(2);
      document.getElementById('printMode').innerText = orderData.mode;
      document.getElementById('printClient').innerText = orderData.customerName || 'Walk-in';
      document.getElementById('printPhone').innerText = orderData.customerPhone || '';
      const discountRow = document.getElementById('printDiscountRow');
      if (discountPercent > 0 && discountAmount > 0) {
        discountRow.style.display = 'inline';
        document.getElementById('printDiscountPercent').innerText = discountPercent.toFixed(1).replace('.0', '');
        document.getElementById('printDiscountAmount').innerText = discountAmount.toFixed(2);
      } else {
        discountRow.style.display = 'none';
      }
      
      let itemsHtml = '';
      orderData.items.forEach(i => {
        itemsHtml += `<div class="receipt-item"><span>${i.qty}x ${i.name}</span><span>${i.price * i.qty}</span></div>`;
      });
      document.getElementById('printItems').innerHTML = itemsHtml;
      
      window.print();
    }

    // --- TABLE SETUP LOGIC ---
    function loadTableSetup() {
      google.script.run.withSuccessHandler(function(tables) {
        const tbody = document.getElementById('tableTableBody');
        tbody.innerHTML = '';
        
        tables.forEach(t => {
          tbody.innerHTML += `<tr>
            <td>${t.name}</td><td>${t.capacity} pax</td>
            <td><span class="status-badge status-${t.status}">${t.status}</span></td>
            <td><button class="btn-secondary" onclick="editTable('${t.id}', '${t.name}', ${t.capacity}, '${t.status}')">Edit</button></td>
          </tr>`;
        });

        const posSelect = document.getElementById('posTableSelect');
        posSelect.innerHTML = '<option value="Takeaway">Takeaway / No Table</option>';
        tables.filter(t => t.status === 'Active').forEach(t => {
          posSelect.innerHTML += `<option value="${t.name}">${t.name} (${t.capacity} pax)</option>`;
        });
      }).getTables();
    }

    function saveTableUI() {
      const table = {
        id: document.getElementById('tableId').value,

    // --- ORDERS LOG ---
    function loadOrders() {
      google.script.run.withSuccessHandler(function(orders) {
        currentOrders = orders;
        renderOrders(orders);
      }).withFailureHandler(function(err) {
        const tbody = document.getElementById('ordersTableBody');
        tbody.innerHTML = '<tr><td colspan="9">Unable to load orders. Please refresh.</td></tr>';
        document.getElementById('ordersSummary').innerText = '';
        showToast(`Orders load failed: ${err.message || err}`, 'info');
      }).getOrders();
    }

    function renderOrders(orders) {
        const tbody = document.getElementById('ordersTableBody');
        tbody.innerHTML = '';
        if (!orders.length) {
          tbody.innerHTML = '<tr><td colspan="9">No orders found for today.</td></tr>';
          document.getElementById('ordersSummary').innerText = 'Showing 0 orders for today.';
          return;
        }
        orders.forEach(o => {
          const isClosed = o.status === 'Closed' || o.status === 'Completed';
          const isCancelled = o.status === 'Cancelled';
          tbody.innerHTML += `<tr>
            <td data-label="ID">${o.id}</td><td data-label="Date">${o.date}</td><td data-label="Client">${o.customerName || 'Walk-in'}</td><td data-label="Phone">${o.customerPhone || '-'}</td><td data-label="Table">${o.table || 'Takeaway'}</td><td data-label="Total">₹${o.total}</td><td data-label="Payment">${o.mode || 'Pending'}</td>
            <td data-label="Status"><span class="status-badge status-${o.status}">${o.status}</span></td>
            <td data-label="Actions">
              <div class="actions-group">
                <button title="Edit Order" class="btn-secondary icon-btn" onclick="editOrder('${o.id}')" ${isClosed || isCancelled ? 'disabled' : ''}>✏️</button>
                <button title="Mark Closed" class="btn-secondary icon-btn" onclick="closeOrder('${o.id}')" ${isClosed || isCancelled ? 'disabled' : ''}>✅</button>
                <button title="Record Payment" class="btn-secondary icon-btn" onclick="openPaymentModal('${o.id}')" ${isCancelled ? 'disabled' : ''}>💳</button>
                <button title="Generate Bill" class="btn-secondary icon-btn" onclick="openBillModal('${o.id}')" ${isCancelled ? 'disabled' : ''}>🧾</button>
                <button title="Cancel Order" class="btn-secondary icon-btn" onclick="cancelOrder('${o.id}')" ${isClosed || isCancelled ? 'disabled' : ''}>❌</button>
              </div>
            </td>
          </tr>`;
        });
        document.getElementById('ordersSummary').innerText = `Showing ${orders.length} order(s) for today.`;
    }

    function filterOrdersList() {
      const q = (document.getElementById('orderSearchInput').value || '').toLowerCase().trim();
      if (!q) {
        renderOrders(currentOrders);
        return;
      }
      const filtered = currentOrders.filter(o => {
        const haystack = `${o.id} ${o.customerName || ''} ${o.customerPhone || ''} ${o.table || ''}`.toLowerCase();
        return haystack.includes(q);
      });
      renderOrders(filtered);
      document.getElementById('ordersSummary').innerText = `Showing ${filtered.length} of ${currentOrders.length} order(s) for today.`;
    }

    }

    function openBillModal(orderId) {
      pendingBillOrderId = orderId;
      document.getElementById('billDiscountPercent').value = 0;
      document.getElementById('billPaymentMode').value = 'Cash';
      document.getElementById('billText').innerText = `Generate printable restaurant bill for order ${orderId}.`;
      document.getElementById('billModal').classList.add('show');
    }

    function closeBillModal() {
      pendingBillOrderId = null;
      document.getElementById('billModal').classList.remove('show');
    }

    function generateOrderBill() {
      if (!pendingBillOrderId) return;
      const order = currentOrders.find(o => o.id === pendingBillOrderId);
      if (!order) {
        closeBillModal();
        return showToast('Order not found for billing.', 'info');
      }
      const discountPercent = Number(document.getElementById('billDiscountPercent').value || 0);
      if (discountPercent < 0 || discountPercent > 100) {
        return showToast('Discount must be between 0 and 100%.', 'info');
      }
      const subtotal = Number(order.total || 0);
      const discountAmount = subtotal * (discountPercent / 100);
      const finalTotal = Math.max(0, subtotal - discountAmount);
      const printPayload = {
        ...order,
        subtotal: subtotal,
        discountPercent: discountPercent,
        discountAmount: discountAmount,
        total: finalTotal,
        mode: document.getElementById('billPaymentMode').value
      };
      printReceipt(printPayload);
      closeBillModal();
      showToast(`Bill generated for ${order.id}.`, 'success');
    }

    }

    function loadExpenses() {
      google.script.run
        .withSuccessHandler(function(exp) {
          const tbody = document.getElementById('expenseTableBody');
          tbody.innerHTML = '';
          if (!exp || exp.length === 0) {
            tbody.innerHTML = '<tr><td colspan="4" style="text-align:center;color:#666;">No expense records found.</td></tr>';
            return;
          }
          exp.forEach(e => {
            tbody.innerHTML += `<tr><td>${e.date || '-'}</td><td>${e.category || '-'}</td><td>${e.desc || '-'}</td><td>₹${e.amount || 0}</td></tr>`;
          });
        })
        .withFailureHandler((err) => {
          showToast(`Unable to load expenses: ${err.message || err}`, 'info');
          const tbody = document.getElementById('expenseTableBody');
          tbody.innerHTML = '<tr><td colspan="4" style="text-align:center;color:#cc0c39;">Failed to load expenses.</td></tr>';
        })
        .getExpenses();
    }

    function saveExpense() {
      const amount = Number(document.getElementById('expAmount').value);
      const exp = {
        category: document.getElementById('expCategory').value,
        amount: amount,
        desc: document.getElementById('expDesc').value.trim()
      };
      if(!amount || amount <= 0) {
        return showToast('Please enter a valid amount greater than 0.', 'info');
      }
      google.script.run
        .withSuccessHandler(() => {
          document.getElementById('expAmount').value = '';
          document.getElementById('expDesc').value = '';
          showToast('Expense saved successfully.', 'success');
          loadExpenses();
        })
        .withFailureHandler((err) => {
          showToast(`Expense save failed: ${err.message || err}`, 'info');
        })
        .saveExpense(exp);
    }
