  <script>
    let globalMenu = [];
    let cart = [];
    let pendingOrder = null;
    let pendingPaymentOrder = null;

    // Navigation
    function nav(pageId, btnEl) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
      document.getElementById(pageId).classList.add('active');
      const labels = {
        pos: ['Point of Sale', 'POS'],
        orders: ['Orders'],
        tables: ['Tables'],
        menu: ['Menu'],
        expenses: ['Expenses']
      };
      document.querySelectorAll('.nav-btn').forEach(btn => {
        if ((labels[pageId] || []).includes(btn.textContent.trim())) btn.classList.add('active');
      });
      if (btnEl) btnEl.classList.add('active');
      document.getElementById('topNav').classList.remove('open');
    }

    function toggleMobileMenu() {
      document.getElementById('topNav').classList.toggle('open');
    }

    function showToast(message, type = 'info') {
      const wrap = document.getElementById('toastWrap');
      const toast = document.createElement('div');
      toast.className = `toast ${type}`;
      toast.innerText = message;
      wrap.appendChild(toast);
      setTimeout(() => {
        toast.style.opacity = '0';
        toast.style.transform = 'translateY(8px)';
      }, 2300);
      setTimeout(() => toast.remove(), 2700);
    }

    // Startup
    window.onload = function() {
      loadPosMenu();
      loadTableSetup();
    };

    // --- POS FUNCTIONS ---
    function loadPosMenu() {
      google.script.run.withSuccessHandler(function(menu) {
        globalMenu = menu;
        const posDiv = document.getElementById('posMenuList');
        posDiv.innerHTML = '';
        menu.filter(m => m.status === 'Active').forEach(item => {
          posDiv.innerHTML += `<div class="menu-item-btn" onclick="addToCart('${item.id}', '${item.name}', ${item.price})">
            <strong>${item.name}</strong><br>₹${item.price}
          </div>`;
        });
      }).getMenu();
    }

    function addToCart(id, name, price) {
      const existing = cart.find(i => i.id === id);
      if (existing) existing.qty++;
      else cart.push({ id, name, price, qty: 1 });
      renderCart();
    }

    function renderCart() {
      const cartDiv = document.getElementById('cartItems');
      cartDiv.innerHTML = '';
      let total = 0;
      cart.forEach((item, index) => {
        total += item.price * item.qty;
        cartDiv.innerHTML += `<li style="display:flex; justify-content:space-between; margin-bottom:8px;">
          <span>${item.qty}x ${item.name}</span>
          <span>₹${item.price * item.qty} <button onclick="cart.splice(${index},1); renderCart();" style="border:none; background:none; color:red; cursor:pointer;">✖</button></span>
        </li>`;
      });
      document.getElementById('cartTotal').innerText = total;
    }

    function processOrder() {
      if (cart.length === 0) return showToast('Cart is empty. Add items before booking.', 'info');
      pendingOrder = {
        items: cart,
        table: document.getElementById('posTableSelect').value,
        total: document.getElementById('cartTotal').innerText,
        mode: 'Pending',
        customerName: document.getElementById('clientName').value || 'Walk-in',
        customerPhone: document.getElementById('clientPhone').value || '-'
      };
      if (pendingOrder.customerPhone !== '-' && !/^\d{10}$/.test(pendingOrder.customerPhone)) {
        return showToast('Please enter a valid 10-digit mobile number.', 'info');
      }
      document.getElementById('confirmText').innerText = `Table: ${pendingOrder.table}`;
      document.getElementById('confirmText2').innerText = `${pendingOrder.customerName} (${pendingOrder.customerPhone}) • Total: ₹${pendingOrder.total}`;
      document.getElementById('confirmModal').classList.add('show');
    }

    function closeConfirmModal() {
      pendingOrder = null;
      document.getElementById('confirmModal').classList.remove('show');
    }

    function confirmAndSaveOrder() {
      if (!pendingOrder) return;
      document.getElementById('confirmModal').classList.remove('show');
      document.querySelector('#pos .btn-primary').innerText = 'Booking...';
      document.querySelector('#pos .btn-primary').disabled = true;
      google.script.run.withSuccessHandler(function(res) {
        printReceipt(res);
        cart = [];
        renderCart();
        document.getElementById('clientName').value = '';
        document.getElementById('clientPhone').value = '';
        document.querySelector('#pos .btn-primary').innerText = 'Book Order';
        document.querySelector('#pos .btn-primary').disabled = false;
        showToast(`Order ${res.id} booked. Set Confirmed/Completed from Orders tab.`, 'success');
        pendingOrder = null;
      }).saveOrder(pendingOrder);
    }

    function printReceipt(orderData) {
      document.getElementById('printId').innerText = orderData.id;
      document.getElementById('printDate').innerText = orderData.date;
      document.getElementById('printTable').innerText = orderData.table;
      document.getElementById('printTotal').innerText = orderData.total;
      document.getElementById('printMode').innerText = orderData.mode;
      document.getElementById('printClient').innerText = orderData.customerName || 'Walk-in';
      document.getElementById('printPhone').innerText = orderData.customerPhone || '';
      
      let itemsHtml = '';
      orderData.items.forEach(i => {
        itemsHtml += `<div class="receipt-item"><span>${i.qty}x ${i.name}</span><span>${i.price * i.qty}</span></div>`;
      });
      document.getElementById('printItems').innerHTML = itemsHtml;
      
      window.print();
    }

    // --- TABLE SETUP LOGIC ---
    function loadTableSetup() {
      google.script.run.withSuccessHandler(function(tables) {
        const tbody = document.getElementById('tableTableBody');
        tbody.innerHTML = '';
        
        tables.forEach(t => {
          tbody.innerHTML += `<tr>
            <td>${t.name}</td><td>${t.capacity} pax</td>
            <td><span class="status-badge status-${t.status}">${t.status}</span></td>
            <td><button class="btn-secondary" onclick="editTable('${t.id}', '${t.name}', ${t.capacity}, '${t.status}')">Edit</button></td>
          </tr>`;
        });

        const posSelect = document.getElementById('posTableSelect');
        posSelect.innerHTML = '<option value="Takeaway">Takeaway / No Table</option>';
        tables.filter(t => t.status === 'Active').forEach(t => {
          posSelect.innerHTML += `<option value="${t.name}">${t.name} (${t.capacity} pax)</option>`;
        });
      }).getTables();
    }

    function saveTableUI() {
      const table = {
        id: document.getElementById('tableId').value,
        name: document.getElementById('tableName').value,
        capacity: document.getElementById('tableCapacity').value,
        status: document.getElementById('tableStatus').value || 'Active'
      };
      if(!table.name) return alert('Table Name is required');
      
      google.script.run.withSuccessHandler(() => {
        document.getElementById('tableId').value = '';
        document.getElementById('tableName').value = '';
        document.getElementById('tableCapacity').value = '';
        loadTableSetup(); 
      }).saveTable(table);
    }

    function editTable(id, name, capacity, status) {
      document.getElementById('tableId').value = id;
      document.getElementById('tableName').value = name;
      document.getElementById('tableCapacity').value = capacity;
      document.getElementById('tableStatus').value = status;
    }

    // --- MENU SETUP ---
    function loadMenuSetup() {
      google.script.run.withSuccessHandler(function(menu) {
        const tbody = document.getElementById('menuTableBody');
        tbody.innerHTML = '';
        menu.forEach(item => {
          tbody.innerHTML += `<tr>
            <td>${item.name}</td><td>${item.category}</td><td>₹${item.price}</td>
            <td><span class="status-badge status-${item.status}">${item.status}</span></td>
            <td><button class="btn-secondary" onclick="editMenu('${item.id}', '${item.name}', '${item.category}', ${item.price}, '${item.status}')">Edit</button></td>
          </tr>`;
        });
      }).getMenu();
    }

    function saveMenu() {
      const item = {
        id: document.getElementById('menuId').value,
        name: document.getElementById('menuName').value,
        category: document.getElementById('menuCategory').value,
        price: document.getElementById('menuPrice').value,
        status: document.getElementById('menuStatus').value || 'Active'
      };
      if(!item.name || !item.price) return alert('Name and Price required');
      
      google.script.run.withSuccessHandler(() => {
        document.getElementById('menuId').value = '';
        document.getElementById('menuName').value = '';
        document.getElementById('menuPrice').value = '';
        loadMenuSetup();
        loadPosMenu();
      }).saveMenuItem(item);
    }

    function editMenu(id, name, cat, price, status) {
      document.getElementById('menuId').value = id;
      document.getElementById('menuName').value = name;
      document.getElementById('menuCategory').value = cat;
      document.getElementById('menuPrice').value = price;
      document.getElementById('menuStatus').value = status;
    }

    // --- ORDERS LOG ---
    function loadOrders() {
      google.script.run.withSuccessHandler(function(orders) {
        const tbody = document.getElementById('ordersTableBody');
        tbody.innerHTML = '';
        orders.forEach(o => {
          const isCompleted = o.status === 'Completed';
          tbody.innerHTML += `<tr>
            <td>${o.id}</td><td>${o.date}</td><td>${o.customerName || 'Walk-in'}</td><td>${o.customerPhone || '-'}</td><td>${o.table || 'Takeaway'}</td><td>₹${o.total}</td><td>${o.mode || 'Pending'}</td>
            <td><span class="status-badge status-${o.status}">${o.status}</span></td>
            <td>
              <select onchange="updateStatus('${o.id}', this.value)" style="width:auto; padding:4px;" ${isCompleted ? 'disabled' : ''}>
                <option value="">Action...</option>
                <option value="Confirmed" ${o.status === 'Confirmed' ? 'selected' : ''}>Confirm</option>
                <option value="Completed">Complete + Payment</option>
                <option value="Cancelled" ${o.status === 'Cancelled' ? 'selected' : ''}>Cancel</option>
              </select>
            </td>
          </tr>`;
        });
      }).getOrders();
    }

    function updateStatus(id, status) {
      if(!status) return;
      if (status === 'Completed') {
        pendingPaymentOrder = id;
        document.getElementById('paymentText').innerText = `Order ${id} is delivered. Record payment to complete it.`;
        document.getElementById('paymentModal').classList.add('show');
        return;
      }
      google.script.run
        .withSuccessHandler(() => {
          showToast(`Order ${id} marked ${status}.`, 'success');
          loadOrders();
        })
        .withFailureHandler((err) => showToast(`Status update failed: ${err.message || err}`, 'info'))
        .updateOrderStatus(id, status);
    }

      pendingPaymentOrder = null;
      document.getElementById('paymentModal').classList.remove('show');

    function loadExpenses() {
      google.script.run.withSuccessHandler(function(exp) {
        const tbody = document.getElementById('expenseTableBody');
        tbody.innerHTML = '';
        exp.forEach(e => {
          tbody.innerHTML += `<tr><td>${e.date}</td><td>${e.category}</td><td>${e.desc}</td><td>₹${e.amount}</td></tr>`;
        });
      }).getExpenses();
    }

    function saveExpense() {
      const amount = Number(document.getElementById('expAmount').value);
      const exp = {
        category: document.getElementById('expCategory').value,
        amount: amount,
        desc: document.getElementById('expDesc').value.trim()
      };
      if(!amount || amount <= 0) {
        return showToast('Please enter a valid amount greater than 0.', 'info');
      }
      google.script.run
        .withSuccessHandler(() => {
          document.getElementById('expAmount').value = '';
          document.getElementById('expDesc').value = '';
          showToast('Expense saved successfully.', 'success');
          loadExpenses();
        })
        .withFailureHandler((err) => {
          showToast(`Expense save failed: ${err.message || err}`, 'info');
        })
        .saveExpense(exp);
    }
