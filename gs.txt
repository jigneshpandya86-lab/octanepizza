const DB_ID = 'PASTE_YOUR_SPREADSHEET_ID_HERE'; 

function doGet() {
  return HtmlService.createHtmlOutputFromFile('Index')
    .setTitle('Octane Pizza POS')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// --- MENU FUNCTIONS ---
function getMenu() {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Menu');
  const data = sheet.getDataRange().getValues();
  data.shift(); 
  return data.map(row => ({ id: row[0], name: row[1], category: row[2], price: row[3], status: row[4] }));
}

function saveMenuItem(item) {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Menu');
  if (item.id) {
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] == item.id) {
        sheet.getRange(i + 1, 2, 1, 4).setValues([[item.name, item.category, item.price, item.status]]);
        break;
      }
    }
  } else {
    const newId = 'M' + new Date().getTime();
    sheet.appendRow([newId, item.name, item.category, item.price, 'Active']);
  }
  return true;
}

// --- TABLE FUNCTIONS ---
function getTables() {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Tables');
  const data = sheet.getDataRange().getValues();
  data.shift(); 
  return data.map(row => ({ id: row[0], name: row[1], capacity: row[2], status: row[3] }));
}

function saveTable(table) {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Tables');
  if (table.id) {
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] == table.id) {
        sheet.getRange(i + 1, 2, 1, 3).setValues([[table.name, table.capacity, table.status]]);
        break;
      }
    }
  } else {
    const newId = 'TBL' + new Date().getTime();
    sheet.appendRow([newId, table.name, table.capacity, 'Active']);
  }
  return true;
}

// --- ORDER FUNCTIONS ---
function getOrders() {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Orders');
  const data = sheet.getDataRange().getValues();
  data.shift();
  // Columns: 0:ID, 1:Date, 2:Table, 3:Items, 4:Total, 5:Mode, 6:Status, 7:Customer Name, 8:Customer Phone
  return data.map(row => ({
    id: row[0],
    date: row[1],
    table: row[2],
    items: JSON.parse(row[3]),
    total: row[4],
    mode: row[5] || 'Pending',
    status: row[6],
    customerName: row[7] || 'Walk-in',
    customerPhone: row[8] || '-'
  })).reverse();
}

function saveOrder(order) {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Orders');
  const newId = 'ORD' + Math.floor(Math.random() * 10000);
  const date = new Date().toLocaleString();
  sheet.appendRow([
    newId,
    date,
    order.table,
    JSON.stringify(order.items),
    order.total,
    'Pending',
    'Booked',
    order.customerName || 'Walk-in',
    order.customerPhone || '-'
  ]);
  return { id: newId, date: date, ...order };
}

function updateOrderStatus(orderId, newStatus) {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Orders');
  const data = sheet.getDataRange().getValues();
  let updated = false;
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] == orderId) {
      sheet.getRange(i + 1, 7).setValue(newStatus); // Column 7 is Status
      updated = true;
      break;
    }
  }
  if (!updated) throw new Error('Order not found: ' + orderId);
  return { ok: true, orderId: orderId, status: newStatus };
}

function completeOrderAndRecordPayment(orderId, paymentMode) {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Orders');
  const data = sheet.getDataRange().getValues();
  let updated = false;
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] == orderId) {
      sheet.getRange(i + 1, 6).setValue(paymentMode); // Column 6 is Payment Mode
      sheet.getRange(i + 1, 7).setValue('Completed'); // Column 7 is Status
      updated = true;
      break;
    }
  }
  if (!updated) throw new Error('Order not found: ' + orderId);
  return { ok: true, orderId: orderId, status: 'Completed', paymentMode: paymentMode };
}

// --- EXPENSE FUNCTIONS ---
function getExpenses() {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Expenses');
  const data = sheet.getDataRange().getValues();
  data.shift();
  return data.map(row => ({ id: row[0], date: row[1], category: row[2], amount: row[3], desc: row[4] })).reverse();
}

function saveExpense(expense) {
  const sheet = SpreadsheetApp.openById(DB_ID).getSheetByName('Expenses');
  const newId = 'EXP' + new Date().getTime();
  const date = new Date().toLocaleDateString();
  sheet.appendRow([newId, date, expense.category, expense.amount, expense.desc]);
  return true;
}
